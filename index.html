function buildPyramid(height = 5, char = '*') {
  let pyramid = '';
  for (let i = 1; i <= height; i++) {
    pyramid += ' '.repeat(height - i);
    pyramid += char.repeat(i * 2 - 1);
    pyramid += '\n';
  }
  return pyramid;
}
